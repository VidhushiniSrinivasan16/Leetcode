public class Solution {
    List<List<Integer>> finalLst=new ArrayList<List<Integer>>();
    public List<List<Integer>> subsets(int[] nums) {
       int n=nums.length;
       for(int i=0;i<(1<<n);i++)
       {
           List<Integer> lst=new ArrayList<Integer>();
           for(int j=0;j<n;j++)
           {
               if((i & (1<<j))>0)
               lst.add(nums[j]);
           }
           finalLst.add(lst);
       }
       return finalLst;
    }
}
